\section{Directional Wrench Tensor and Endogenous Graph}
\label{sec:model}

We consider $N$ robots indexed by $i$, $M$ loads indexed by $k$, and $H$ contact
slots per load indexed by $h$. Each robot $i$ chooses a mixed action
$x_i\in\Delta_i$ over its load/slot, relay, and idle options; we write $x_{ikh}$
for the share it commits to slot $h$ of load $k$ and $x_{ir}$ for the share it
commits to role $r$.

\subsection{Directional wrench-capacity tensor}
Let $U_i\subset\R^2$ be the compact convex set of planar forces robot $i$ can
exert, $A_{kh}\in\R^{3\times2}$ the map from a contact force to the planar wrench
(force-$x$, force-$y$, torque) it induces on load $k$, and $\eta_{k\ell}\in\R^3$
a dual wrench direction. With $h_{U_i}(\cdot)$ the support function of $U_i$,
define the \emph{directional wrench-capacity tensor}
\begin{equation}
\Wten_{ikh\ell} \;=\; h_{U_i}\!\big(A_{kh}^{\top}\eta_{k\ell}\big),
\qquad \Wten\in\R^{N\times M\times H\times P}.
\label{eq:tensor}
\end{equation}
Each entry is the largest force-projection robot $i$ can supply along the pulled-back
direction $A_{kh}^{\top}\eta_{k\ell}$ at slot $h$. The aggregated directional
capacity and the projected demand are
\begin{equation}
s_{k\ell}(x)=\sum_{i,h}\Wten_{ikh\ell}\,x_{ikh},
\qquad
d_{k\ell}=\eta_{k\ell}^{\top}\wdem_k ,
\label{eq:cap-dem}
\end{equation}
where $\wdem_k\in\R^3$ is the wrench demanded by load $k$. The coalition is
\emph{wrench-feasible} for load $k$ when
\begin{equation}
s_{k\ell}(x)\ge d_{k\ell}\quad\forall\ell .
\label{eq:wrench-feasible}
\end{equation}

\begin{assumption}[Complete facet set]\label{as:facets}
For each load $k$ the directions $\{\eta_{k\ell}\}_{\ell=1}^{P}$ contain the outer
normals of every facet of the reachable wrench set
$\Wten(C_k)=\bigoplus_{i,h}A_{kh}\,U_i$ (a Minkowski sum / zonotope).
\end{assumption}

\begin{proposition}[Tensorized membership]\label{prop:membership}
Under Assumption~\ref{as:facets}, \eqref{eq:wrench-feasible} holds for all $\ell$
iff $\wdem_k\in\Wten(C_k)$.
\end{proposition}
\begin{proof}
$\Wten(C_k)$ is compact convex, so by the supporting-hyperplane theorem
$\wdem_k\in\Wten(C_k)$ iff $\eta^{\top}\wdem_k\le h_{\Wten(C_k)}(\eta)$ for every
outer normal $\eta$. Support functions are additive over Minkowski sums and
positively homogeneous, so
$h_{\Wten(C_k)}(\eta_{k\ell})=\sum_{i,h}h_{U_i}(A_{kh}^{\top}\eta_{k\ell})
=\sum_{i,h}\Wten_{ikh\ell}$ at $x\equiv1$, and $s_{k\ell}(x)$ is exactly this sum
weighted by the committed shares. Assumption~\ref{as:facets} makes the finite set
$\{\eta_{k\ell}\}$ sufficient.
\end{proof}
A scalar capacity $c_i$ cannot express differences in force limit, lever arm,
slot access, or longitudinal/lateral/torque utility that \eqref{eq:tensor} keeps
in separate channels.

\subsection{Endogenous, role-induced communication graph}
Let $e$ index a potential communication link between regions, and let
$\mathcal{C}_{ire}\ge0$ be robot $i$'s contribution to link $e$ when it plays role
$r$ (e.g.\ occupying a relay site). Edge weights and the induced graph Laplacian
are then affine in the decisions,
\begin{equation}
a_e(x)=\sum_{i,r}\mathcal{C}_{ire}\,x_{ir},
\qquad
\Lap(x)=\sum_e a_e(x)\,L_e ,
\label{eq:laplacian}
\end{equation}
with $L_e\succeq0$ the elementary Laplacian of link $e$.

Let $Q\in\R^{V\times(V-1)}$ be an orthonormal basis of $\mathbf 1^{\perp}$. The
information constraint is the linear matrix inequality (LMI)
\begin{equation}
Q^{\top}\Lap(x)\,Q\;\succeq\;\sigmastar I,
\qquad\Longleftrightarrow\qquad
\lamtwo(\Lap(x))\ge\sigmastar ,
\label{eq:lmi}
\end{equation}
which is affine in $x$; its Lagrange dual is a positive-semidefinite matrix
price $Z\succeq0$. At a point where the active eigenvalue is simple, $Z$ collapses
to the rank-one $Z=\pi\,v v^{\top}$ with $v$ the unit Fiedler vector, recovering a
scalar connectivity price $\pi$ (used in the algorithm of
Section~\ref{sec:algorithm}). Working with \eqref{eq:lmi} rather than $\lamtwo$
directly avoids the nonsmoothness of the eigenvalue at multiplicity.

\begin{lemma}[Convex information set]\label{lem:convex}
$x\mapsto\lamtwo(\Lap(x))$ is concave, hence the LMI region
$\{x:Q^{\top}\Lap(x)Q\succeq\sigmastar I\}=\{x:\lamtwo(\Lap(x))\ge\sigmastar\}$ is convex.
\end{lemma}
\begin{proof}
By Fiedler~\cite{fiedler1973},
$\lamtwo(L)=\min_{\|v\|=1,\,v\perp\mathbf 1} v^{\top}Lv$ is a pointwise minimum of
functions linear in $L$, so it is concave in $L$; since $L(x)$ is affine in $x$,
the composition is concave and its super-level set is convex~\cite{ghosh2006growing}.
\end{proof}

The information threshold comes from the Nash-seeking dynamics: consensus must
contract estimation error faster than cross-coupling injects it, giving
\begin{equation}
\boxed{\;\sigmastar=\dfrac{\vartheta^{2}}{c\,\mF}\;}
\label{eq:sigma-star}
\end{equation}
with $c$ the consensus gain, $\mF$ the strong-monotonicity constant of the
pseudo-gradient, and $\vartheta$ the cross-coupling magnitude. We write $\mF$,
not $\mu$, to keep the monotonicity constant distinct from the wrench price.
