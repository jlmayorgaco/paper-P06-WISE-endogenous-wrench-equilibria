\section{Self-Sustaining Equilibria}
\label{sec:wise}

\subsection{Productive value and the composition face}
The team's productive value couples through aggregates only,
\begin{equation}
V(x)=\sum_k F_k\!\big(s_k(x)\big)-\sum_{\tau,a}g_{\tau a}x_{\tau a},
\label{eq:value}
\end{equation}
with $F_k$ strictly concave and non-decreasing (a canonical choice is the deficit
value $F_k(s_k)=-\tfrac12\|[d_k-s_k]_+\|_{Q_k}^2$, $Q_k\succ0$) and $g_{\tau a}\ge0$
the action cost. This is a \emph{constrained potential population game}: the
type-action payoff is $\partial V/\partial x_{\tau a}$, so $V$ is an exact
potential by construction~\cite{sandholm2010population,monderer1996potential}. We
do not claim novelty here---the game is identical-interest---the novelty is what
$V$ leaves undetermined.

Let $y=Bx$ collect the aggregates $s(x)$ appearing in \eqref{eq:value}, with $B$
linear, and let $X_{\mathrm f}=\{x\in\mathcal X:s_{k\ell}(x)\ge d_{k\ell}\}$ be the
wrench-feasible set. Write $V(x)=\Phi_0(Bx)-\langle g,x\rangle$.

\begin{theorem}[Composition degeneracy]\label{thm:degeneracy}
Let $\Phi_0$ be strictly concave and $g$ lie in the row space of $B$
(cost expressible through aggregates). Then the productively optimal aggregate
$y^{\star}=Bx^{\star}$ is unique, but the optimal set
\begin{equation}
\Emanifold=\{x\in X_{\mathrm f}:Bx=y^{\star}\}
\label{eq:manifold}
\end{equation}
is a polytope of dimension $(T-1)(K_{\mathrm{act}}-1)$, where $K_{\mathrm{act}}$ is
the number of active (load,\,slot) columns.
\end{theorem}
\begin{proof}
Strict concavity of $\Phi_0$ makes the maximiser $y^{\star}$ of
$\max_y\{\Phi_0(y)-\text{(aggregate cost)}:y\in B X_{\mathrm f}\}$ unique. $\Emanifold$
is the preimage $B^{-1}(y^{\star})\cap X_{\mathrm f}$, an affine slice of the product
of type simplices with fixed row sums $m_\tau$ and fixed column aggregates
$y^{\star}$---a transportation polytope, whose dimension is
$(T-1)(K_{\mathrm{act}}-1)$~\cite{sandholm2010population}. Since $g$ acts through
$B$, every $x\in\Emanifold$ attains the same $V$.
\end{proof}
Classical theory thus fixes \emph{how much} wrench each load receives, not
\emph{which types} provide it. The face $\Emanifold$ is the freedom we exploit.

\subsection{The WISE refinement}
Place the endogenous Laplacian \eqref{eq:laplacian} on $\Emanifold$.

\begin{definition}[WISE]\label{def:wise}
A profile $x^{\star}\in\Emanifold$ is a \emph{Wrench- and
Information-Self-Sustaining Equilibrium} if $\lamtwo(\Lap(x^{\star}))\ge\sigmastar$,
i.e.\ $x^{\star}\in\Ess:=\Emanifold\cap\{x:\lamtwo(\Lap(x))\ge\sigmastar\}$.
\end{definition}

\begin{theorem}[Existence by selection]\label{thm:existence}
Let $\Lammax=\max_{x\in\Emanifold}\lamtwo(\Lap(x))$. Then
\begin{equation}
\Ess\neq\varnothing\iff\Lammax\ge\sigmastar,
\label{eq:exist}
\end{equation}
and $\Ess$ is convex. When $\dim\Emanifold>0$ and
$\min_{\Emanifold}\lamtwo<\sigmastar\le\Lammax$, the refinement is \emph{strict}:
some productively optimal compositions self-sustain and others do not.
\end{theorem}
\begin{proof}
$\lamtwo\!\circ\!\Lap$ is concave (Lemma~\ref{lem:convex}) and continuous on the
compact convex $\Emanifold$, so it attains its maximum $\Lammax$; $\Ess$ is
nonempty iff $\Lammax\ge\sigmastar$. $\Ess$ is the intersection of the polytope
$\Emanifold$ with a convex super-level set, hence convex. Strictness is immediate
when the threshold falls strictly inside the range of $\lamtwo$ over $\Emanifold$.
\end{proof}
Selection within a degenerate Nash face is stronger than appending a constraint to
a strictly concave program: without Theorem~\ref{thm:degeneracy} there would be a
single composition and nothing to select.

\subsection{The self-defeat threshold, derived}
The distributed dynamics run a strategy-revision loop and a consensus estimator of
the aggregates over $\Lap(x)$. Linearising about $x^{\star}$, with $m_F$ the
strong-monotonicity constant of the revision protocol, $c$ the consensus gain, and
$\vt$ the cross-coupling by which stale aggregate estimates perturb revision, the
coupled error $e=(e_x,e_\eta)$ obeys
\begin{equation}
\dot e=J\,e,\qquad
J=\begin{bmatrix}-m_F & \vt\\[1mm] \vt & -c\,\lamtwo(\Lap(x^{\star}))\end{bmatrix}.
\label{eq:jac}
\end{equation}

\begin{theorem}[Exact stability threshold]\label{thm:stability}
$x^{\star}$ is locally exponentially stable under \eqref{eq:jac} iff
\begin{equation}
c\,m_F\,\lamtwo(\Lap(x^{\star}))>\vt^{2},
\quad\text{i.e.}\quad
\lamtwo>\sigmastar:=\frac{\vt^{2}}{c\,m_F}.
\label{eq:threshold}
\end{equation}
Below the threshold $J$ has a positive real eigenvalue and $x^{\star}$ is unstable.
\end{theorem}
\begin{proof}
$\operatorname{tr}J=-(m_F+c\lamtwo)<0$ always, and
$\det J=c\,m_F\,\lamtwo-\vt^{2}$. A real $2\times2$ matrix is Hurwitz iff
$\operatorname{tr}J<0$ and $\det J>0$; thus stability $\Leftrightarrow\det J>0
\Leftrightarrow$ \eqref{eq:threshold}. If $\det J<0$ the eigenvalues are real of
opposite sign, giving an unstable manifold.
\end{proof}
This grounds $\sigmastar$ in the dynamics of the very estimator that computes the
equilibrium, and matches the existence condition~\eqref{eq:exist}: a productively
optimal composition is dynamically viable iff its induced connectivity clears the
same $\sigmastar$.

\subsection{Selection, prices, and the cost of endogeneity}
\begin{proposition}[Infinitesimal-reward selection]\label{prop:selection}
For $V_\varepsilon(x)=V(x)+\varepsilon\,\lamtwo(\Lap(x))$, $\varepsilon>0$, every
maximiser $x_\varepsilon$ over $X_{\mathrm f}$ satisfies $Bx_\varepsilon\to y^{\star}$
and, along a subsequence, $x_\varepsilon\to\arg\max_{x\in\Emanifold}\lamtwo(\Lap(x))$
as $\varepsilon\downarrow0$.
\end{proposition}
\begin{proof}
$V$ is upper bounded on the compact $X_{\mathrm f}$ with optimal set $\Emanifold$
(Theorem~\ref{thm:degeneracy}); $\lamtwo\!\circ\!\Lap$ is continuous. By the
Tikhonov/$\Gamma$-selection principle, maximisers of $V+\varepsilon\lamtwo$ converge
as $\varepsilon\downarrow0$ to maximisers of $\lamtwo$ over $\arg\max V=\Emanifold$.
\end{proof}
An infinitesimal information value breaks the composition degeneracy toward the
most-connected profile without moving the productive optimum---an exact
equilibrium refinement.

\begin{corollary}[Two certification prices]\label{cor:prices}
At a WISE solving $\max_{x\in\Ess}V$, let $\mu^{\star}_k=\nabla F_k(s_k^{\star})$ be
the wrench price and $Z^{\star}\succeq0$ the dual of the LMI~\eqref{eq:lmi}, with
$Z^{\star}=\pi^{\star}vv^{\top}$ at a simple minimiser. Stationarity gives the
type-action value
\begin{equation}
q^{\star}_{\tau a}=\underbrace{\textstyle\sum_{k,\ell}\mu^{\star}_{k\ell}\Wmap_{\tau a,k\ell}}_{\text{wrench}}
-\,g_{\tau a}
+\underbrace{\pi^{\star}\,\partial\lamtwo/\partial x_{\tau a}}_{\text{connectivity}},
\label{eq:prices}
\end{equation}
with $\pi^{\star}(\lamtwo-\sigmastar)=0$. All terms of $\nabla V$ are present:
$\mu^{\star}=\nabla F_k$ is the single wrench price, not a separate multiplier.
\end{corollary}

We measure the price of endogeneity \emph{additively}, avoiding the pathologies of
a potential ratio:
\begin{equation}
\Delta_{\mathrm{endo}}
= V(x^{\star}_{\mathrm{exo}})-\max_{x\in\Ess}V(x)\;\ge0,
\label{eq:gap}
\end{equation}
the productive value forgone to remain self-sustaining. $\Delta_{\mathrm{endo}}=0$
exactly when $\Lammax$ is attained at a productively optimal composition---when
radius heterogeneity suffices to bridge the gap for free.

\begin{remark}[Integer recovery]\label{rem:integer}
Rounding the fluid optimum $x^{\star}$ to integer assignments preserves wrench
feasibility and $\lamtwo\ge\sigmastar$ whenever the margins
$m_W=\min_{k,\ell}(s_{k\ell}-d_{k\ell})$ and $m_I=\lamtwo-\sigmastar$ exceed
rounding constants $\kappa_W,\kappa_I=O(1/N)$ set by the granularity.
\end{remark}
